#This code consists of two functions. 
#The first function calculates the total amount of USD coins you have and provides the result. 
#The second function determines the amount that the Cashier will need to give you back as change.

total =0

def pennies(x):
    return x * .01
    
    
def quarters(x):
    return x * .25
    
    
def nickels(x):
    return x * .5
    
    
def dimes(x):
    return x * .1
   

menu_options = ('P', 'Q' , 'N', 'D', 'E')

product_price = int(input("Give me the product's price.\nIf there is no product then enter 0"))

while True:
    print ("**MENU**")
    print ("P for Pennies")
    print ("Q for quarters")
    print ("N for nickels")
    print ("D for dimes")
    print ("E to calculate your total")
    
    print()
    user_input = input("Enter an option")
    
    if user_input in menu_options: 
    
        if user_input == 'P':
            money = int(input("How many Pennies do you have?"))
            total+= pennies(int (money))
            
        elif user_input == 'Q':
            money = int(input("How many quarters do you have?"))
            total+=quarters(int (money))
            
        elif user_input == 'D':
            money = int(input("How many dimes do you have?"))
            total+=dimes(int (money))
            
        elif user_input == 'N':
            money = int(input("How many nickels do you have?"))
            total+=nickels(int (money))
        
        elif user_input == 'E':
            print()
            Change = product_price - total
            print("You have a total of ", total , " Dollars")
            
            if Change <0 and product_price != 0:
                print("we owe ", round(Change,2))
            break
    
    
    else:
        print("**Option not available**")
